\section*{Chapter 1: Sampling \& Descriptive Statistics}
\textbf{Sample Mean}: $\bar{X} = \frac{1}{n}\sum X_i$. \textbf{Variance}: $s^2 = \frac{\sum X_i^2 - n\bar{X}^2}{n-1}$. \\
\textbf{Median}: Position $\frac{n+1}{2}$. \textbf{IQR}: $Q_3 - Q_1$. \\
\textbf{Outlier Fences}: $[Q_1 - 1.5IQR, Q_3 + 1.5IQR]$. \\
\textbf{Linear Trans}: $Y = aX+b \implies \bar{Y} = a\bar{X}+b, s_Y = |a|s_X, s_Y^2 = a^2s_X^2$. \\
\textbf{Z-score}: $z = (x-\bar{x})/s$. \textbf{CV}: $s/\bar{x}$.

\section*{Chapter 2: Probability Foundations}
\textbf{Addition}: $P(A \cup B) = P(A) + P(B) - P(A \cap B)$. \\
\textbf{Conditional}: $P(A|B) = P(A \cap B)/P(B)$. \\
\textbf{Independence}: $P(A \cap B) = P(A)P(B) \iff P(A|B) = P(A)$. \\
\textbf{Total Prob}: $P(B) = \sum P(B|A_i)P(A_i)$. \\
\textbf{Bayes}: $P(A_k|B) = \frac{P(B|A_k)P(A_k)}{\sum P(B|A_i)P(A_i)}$. \\
\textbf{Counting}: $P^n_k = \frac{n!}{(n-k)!}$, $\binom{n}{k} = \frac{n!}{k!(n-k)!}$.

\subsection*{Random Variables \& Moments}
\begin{itemize}
    \item \textbf{Discrete}: $E(X) = \sum x p(x)$, $V(X) = \sum x^2 p(x) - \mu^2$
    \item \textbf{Continuous}: $E(X) = \int x f(x)dx$, $V(X) = \int x^2 f(x)dx - \mu^2$
    \item \textbf{Chebyshev}: $P(|X-\mu| \geq k\sigma) \leq 1/k^2$
    \item \textbf{Linear}: $E(aX+b) = a\mu_X + b$, $V(aX+b) = a^2\sigma_X^2$
    \item \textbf{Sample Mean}: $\mu_{\bar{X}} = \mu$, $\sigma_{\bar{X}} = \sigma/\sqrt{n}$
\end{itemize}

\textbf{Covariance and Correlation}
\begin{itemize}
    \item $\text{Cov}(X,Y) = E(XY) - \mu_X\mu_Y$
    \item $\rho_{X,Y} = \frac{\text{Cov}(X,Y)}{\sigma_X\sigma_Y}$, $-1 \leq \rho \leq 1$
    \item \textbf{Indep} $\implies \text{Cov}=0$. \textbf{Non-Indep Var}: $V(X+Y) = \sigma_X^2 + \sigma_Y^2 + 2\text{Cov}(X,Y)$
\end{itemize}

\section*{Chapter 3: Propagation of Error}
\textbf{Uncertainty Definition}: $\sigma$ (Std Dev). \\
\textbf{General Formula (Independent)}: $\sigma_U = \sqrt{\sum (\frac{\partial U}{\partial X_i})^2 \sigma_{X_i}^2}$. \\
\textbf{General Formula (Dependent)}: $\sigma_U \leq \sum |\frac{\partial U}{\partial X_i}| \sigma_{X_i}$ (Conservative). \\
\textbf{Power Rule Shortcut}: $U = X^a Y^b \implies \frac{\sigma_U}{U} = \sqrt{(a\frac{\sigma_X}{X})^2 + (b\frac{\sigma_Y}{Y})^2}$. \\
\textbf{Calculus Help}: $\frac{d}{dx}(\ln x) = \frac{1}{x}$, $\frac{d}{dx}(\frac{c}{x}) = -\frac{c}{x^2}$.

\section*{Chapter 4: Common Distributions}
\textbf{Binomial}: $P(x) = \binom{n}{x}p^x q^{n-x}$. $\mu=np, \sigma^2=npq$. \\
\textbf{Poisson}: $P(x) = \frac{e^{-\lambda}\lambda^x}{x!}$. $\mu=\lambda, \sigma^2=\lambda$. \\
\textbf{Exponential}: $f(x) = \lambda e^{-\lambda x}$. $E(X)=1/\lambda, V(X)=1/\lambda^2$. \\
\textbf{Normal $N(\mu, \sigma^2)$}: \textbf{TRAP}: Use $\sigma = \sqrt{\sigma^2}$. \\
\textbf{CLT}: If $n \geq 30$, $\bar{X} \sim N(\mu, \sigma^2/n)$. \\
\textbf{Approximations}:
\begin{itemize}
    \item \textbf{Pois to Binom}: $n \ge 100, p \le 0.05, \lambda=np$.
    \item \textbf{Norm to Binom}: $np, nq > 10$. $\mu=np, \sigma = \sqrt{npq}$.
    \item \textbf{Continuity Correction}: Shift by 0.5 (e.g., $P(X \ge 5) \to P(X_{norm} \ge 4.5)$).
\end{itemize}

\section*{Chapter 5: Confidence Intervals}
\textbf{Form}: $\text{Point Est} \pm (\text{Crit Val} \times \text{Std Error})$. \\
\textbf{One Mean}: $\bar{X} \pm t_{n-1, \alpha/2} (s/\sqrt{n})$. (Use $z$ if $\sigma$ known). \\
\textbf{One Prop}: $\tilde{p} \pm z_{\alpha/2} \sqrt{\tilde{p}(1-\tilde{p})/\tilde{n}}$ (Use $X+2, n+4$). \\
\textbf{Welch's $df$ ($\nu$)}: $\nu = \lfloor \frac{(v_1+v_2)^2}{\frac{v_1^2}{n_1-1} + \frac{v_2^2}{n_2-1}} \rfloor$ where $v_i = s_i^2/n_i$. \\
\textbf{Pooled $s_p^2$}: $\frac{(n_1-1)s_1^2 + (n_2-1)s_2^2}{n_1+n_2-2}$. Use if variances equal. \\
\textbf{Prediction Interval}: $\bar{X} \pm t_{n-1, \alpha/2} s \sqrt{1 + 1/n}$. \\
\textbf{Tolerance Interval}: $\bar{X} \pm k_{n, \alpha, \gamma} s$.

\section*{Chapter 6: Hypothesis Testing}
\textbf{Errors}: $\alpha = P(\text{Type I}) = P(\text{Rej } H_0 | H_0 \text{ true})$. \\
$\beta = P(\text{Type II}) = P(\text{Fail Rej } H_0 | H_1 \text{ true})$. \\
\textbf{Power}: $1-\beta = P(\text{Rej } H_0 | H_1 \text{ true})$. \\
\textbf{Test Statistics}:
\begin{itemize}
    \item \textbf{One Mean}: $t = (\bar{X}-\mu_0)/(s/\sqrt{n})$
    \item \textbf{One Prop}: $z = (\hat{p}-p_0)/\sqrt{p_0 q_0/n}$ (\textbf{NOTE}: Use $p_0$ in denom)
    \item \textbf{Two Means}: $z$ or $t = \frac{(\bar{X}-\bar{Y})-\Delta_0}{SE}$
    \item \textbf{Two Props}: $z = \frac{\hat{p}_1-\hat{p}_2}{\sqrt{\hat{p}_{pool}\hat{q}_{pool}(1/n_1+1/n_2)}}$
\end{itemize}

\textbf{Power Analysis Calculation} ($H_1: \mu = \mu_1 > \mu_0$):
\begin{enumerate}
    \item Find critical value $\bar{x}_{crit} = \mu_0 + z_{\alpha}\sigma/\sqrt{n}$.
    \item Find $\beta = P(\bar{X} < \bar{x}_{crit} | \mu = \mu_1) = P(Z < \frac{\bar{x}_{crit}-\mu_1}{\sigma/\sqrt{n}})$.
    \item Sample Size for Power: $n = \frac{(z_\alpha + z_\beta)^2 \sigma^2}{(\mu_1 - \mu_0)^2}$.
\end{enumerate}

\section*{Categorical Data \& Variances}
\textbf{F-Test (Two Variances)}: $F = s_1^2 / s_2^2$. $df_1 = n_1-1, df_2 = n_2-1$. \\
\textbf{Chi-Square GoF}: $\chi^2 = \sum (O_i - E_i)^2/E_i$, $df=k-1-\text{params}$. \\
\textbf{Chi-Square Indep}: $E_{ij} = \frac{(R_i C_j)}{n}$, $df = (r-1)(c-1)$.

\section*{Calculus \& Moment Cheat Sheet}
$\int x e^{-ax} dx = \frac{-e^{-ax}}{a^2}(ax+1)$. \\
$\int x^2 e^{-ax} dx = \frac{-e^{-ax}}{a^3}(a^2x^2 + 2ax + 2)$. \\
\textbf{MGF}: $M_X(t) = E[e^{tX}]$. $E(X^k) = M_X^{(k)}(0)$. \\
\textbf{Integration by Parts}: $\int u dv = uv - \int v du$.

\section*{Strategy: Which Test to Use?}
\begin{tabular}{|l|l|}
\hline
\textbf{Condition} & \textbf{Test} \\ \hline
1 Mean, $\sigma$ Unknown & 1-sample $t$-test \\ \hline
2 Means, Indep, $s_1 \approx s_2$ & Pooled $t$-test \\ \hline
2 Means, Indep, $s_1 \neq s_2$ & Welch $t$-test \\ \hline
Paired Data (Before/After) & Paired $t$-test ($D = X-Y$) \\ \hline
Comparing 2 Variances & F-test \\ \hline
Categorical Counts & Chi-Square test \\ \hline
\end{tabular}

\section*{Exam Day Trap Checklist}
\begin{itemize}
    \item \textbf{Variance vs Std Dev}: Did I square $s$ or take $\sqrt{s^2}$?
    \item \textbf{Denominator}: Did I use $\sigma$ or $\sigma/\sqrt{n}$ for the mean?
    \item \textbf{P-values}: Did I double the tail area for $\neq$ tests?
    \item \textbf{Degrees of Freedom}: Is it $n-1$, $n_1+n_2-2$, or Welch $\nu$?
    \item \textbf{Rounding}: Always round sample size $n$ \textbf{UP} to the next integer.
    \item \textbf{Proportion Power}: Use pooled $\hat{p}$ for $H_0$, but individual $\hat{p}$ for CI.
\end{itemize}
